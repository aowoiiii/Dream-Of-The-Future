   function jx-switch(e){
	   var jx-imgs=document.getElementById("jx-switch-btn").getElementsByTagName("button");
	   var buttons=document.getElementById("jx-switch-img").getElementsByTagName("img");
	   var butSize=buttons.length;
	   for(var i=0;i<butSize;i++){
		   
		   if(buttons[i]==e){
			    
			  jx-imgs[i].className="img-responsive img-show";
		   }else{
			  
			  jx-imgs[i].className="img-responsive img-hide";
		   }
	   }
   }
